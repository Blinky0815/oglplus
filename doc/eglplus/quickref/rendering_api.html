Rendering API

#include <eglplus/rendering_api.hpp> 

enum class RenderingAPI : EGLenum
{
	OpenGL    = EGL_OPENGL_API,
	OpenGL_ES = EGL_OPENGL_ES_API,
	OpenVG    = EGL_OPENVG_API,
	None      = EGL_NONE
};

template <>
Range<RenderingAPI> EnumValueRange<RenderingAPI>(void);

StrCRef EnumValueName(RenderingAPI);

Boolean BindAPI(RenderingAPI api); 

RenderingAPI QueryAPI(void);

Binds the specified rendering api. See eglBindAPI.

Queries the currently bound rendering API. See eglQueryAPI.
