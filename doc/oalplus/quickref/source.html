Sources

#include <oalplus/source.hpp>

Common source operations 

namespace oalplus {

template <>
class ObjectOps<tag::DirectState, tag::Source>
 : public SourceName
{
public:
	void Play(void); 
	void Pause(void); 
	void Stop(void); 
	void Rewind(void); 

	SourceState State(void) const; 

	void Relative(bool value); 
	bool Relative(void) const; 

	void Type(SourceType type); 
	SourceType Type(void) const; 

	void Looping(bool value); 
	bool Looping(void) const; 

	void Buffer(const BufferName& buffer); 
	void DetachBuffers(void); 
	void QueueBuffers(const Sequence<BufferName>& buffers); 
	void UnqueueBuffers(const Sequence<BufferName>& buffers);

Starts the audio playback. See alSourcePlay.

Pauses the audio playback. See alSourcePause.

Stops the audio playback. See alSourceStop.

Rewinds the audio track. See alSourceRewind.

Returns the source state. See alGetSourceiv, GL_SOURCE_STATE.

Sets the source to relative or absoulte state. See alSourcei, GL_SOURCE_RELATIVE.

Returns true if the source is relative. See alGetSourceiv, GL_SOURCE_RELATIVE.

Sets the source type. See alSourcei, GL_SOURCE_TYPE.

Returns the source type. See alGetSourceiv, GL_SOURCE_TYPE.

Sets the looping mode. See alSourcei, GL_LOOPING.

Returns true if the source is in looping mode. See alGetSourceiv, GL_SOURCE_LOOPING.

Assigns an audio buffer to the source. See alSourcei, GL_BUFFER.

Detaches all queued buffers from the source. See alSourcei, GL_BUFFER.

Enqueues multiple buffers to the source. See alSourceQueueBuffers.

Enqueues multiple buffers to the source. See alSourceUnqueueBuffers. 

	void Gain(ALfloat value); 
	ALfloat Gain(void) const; 

	void MinGain(ALfloat value); 
	ALfloat MinGain(void) const; 

	void MaxGain(ALfloat value); 
	ALfloat MaxGain(void) const; 

	void ReferenceDistance(ALfloat value); 
	ALfloat ReferenceDistance(void) const; 

	void RolloffFactor(ALfloat value); 
	ALfloat RolloffFactor(void) const; 

	void MaxDistance(ALfloat value); 
	ALfloat MaxDistance(void) const; 

	void Pitch(ALfloat value); 
	ALfloat Pitch(void) const;

Sets the value of gain. See alSourcef, GL_GAIN.

Returns the value of gain. See alGetSourcefv, GL_GAIN.

Sets the minimal value of gain. See alSourcef, GL_MIN_GAIN.

Returns the minimal value of gain. See alGetSourcefv, GL_MIN_GAIN.

Sets the maximal value of gain. See alSourcef, GL_MAX_GAIN.

Returns the maximal value of gain. See alGetSourcefv, GL_MAX_GAIN.

Sets the reference distance. See alSourcef, GL_REFERENCE_DISTANCE.

Returns the reference distance. See alGetSourcefv, GL_REFERENCE_DISTANCE.

Sets the value of the rolloff factor. See alSourcef, GL_ROLLOFF_FACTOR.

Returns the value of the rolloff factor. See alGetSourcefv, GL_ROLLOFF_FACTOR.

Sets the value of max distance used with some distance models. See alSourcef, GL_MAX_DISTANCE.

Returns the value of max distance used with some distance models. See alGetSourcefv, GL_MAX_DISTANCE.

Sets the value of pitch. See alSourcef, GL_PITCH.

Returns the value of pitch. See alGetSourcefv, GL_PITCH. 

	void Position(const __Vec3f& dir); 
	void Position(ALfloat x, ALfloat y, ALfloat z);
	__Vec3f Position(void) const; 

	void Velocity(const __Vec3f& dir); 
	void Velocity(ALfloat x, ALfloat y, ALfloat z);
	__Vec3f Velocity(void) const; 

	void Direction(const __Vec3f& dir); 
	void Direction(ALfloat x, ALfloat y, ALfloat z);
	__Vec3f Direction(void) const; 

	void ConeInnerAngle(__Anglef angle); 
	__Anglef ConeInnerAngle(void) const; 

	void ConeOuterAngle(__Anglef angle); 
	__Anglef ConeOuterAngle(void) const; 

	void ConeOuterGain(ALfloat value); 
	ALfloat ConeOuterGain(void) const; 

	void SecOffset(ALfloat value); 
	void SampleOffset(ALfloat value); 
	void ByteOffset(ALfloat value); 
};

Sets the position of the source. See alSourcefv, GL_POSITION.

Returns the position of the source. See alGetSourcefv, GL_POSITION.

Sets the velocity vector of the source. See alSourcefv, GL_VELOCITY.

Returns the velocity vector of the source. See alGetSourcefv, GL_VELOCITY.

Sets the direction vector of the source. See alSourcefv, GL_DIRECTION.

Returns the direction vector of the source. See alGetSourcefv, GL_DIRECTION.

Sets the sound cone's inner angle. See alSourcef, GL_CONE_INNER_ANGLE.

Returns the sound cone's inner angle. See alGetSourcefv, GL_CONE_INNER_ANGLE.

Sets the sound cone's outer angle. See alSourcef, GL_CONE_OUTER_ANGLE.

Returns the sound cone's outer angle. See alGetSourcefv, GL_CONE_OUTER_ANGLE.

Sets the sound cone's outer gain value. See alSourcef, GL_CONE_OUTER_GAIN.

Returns the sound cone's outer gain value. See alGetSourcefv, GL_CONE_OUTER_GAIN.

Sets the Second-offset value. See alSourcef, GL_SEC_OFFSET.

Sets the sample-offset value. See alSourcef, GL_SAMPLE_OFFSET.

Sets the byte-offset value. See alSourcef, GL_BYTE_OFFSET.
