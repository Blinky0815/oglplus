Source state

#include <oalplus/source_state.hpp> 

namespace oalplus {

enum class SourceState : ALenum
{
	Initial = AL_INITIAL,
	Playing = AL_PLAYING,
	Paused  = AL_PAUSED,
	Stopped = AL_STOPPED
};

template <>
Range<SourceState> EnumValueRange<SourceState>(void);

StrCRef EnumValueName(SourceState);
