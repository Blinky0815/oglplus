#include <oglplus/string_query.hpp> 

namespace oglplus {

enum class StringQuery : GLenum
{
	Renderer               = GL_RENDERER,
	Vendor                 = GL_VENDOR,
	Version                = GL_VERSION,
	ShadingLanguageVersion = GL_SHADING_LANGUAGE_VERSION
};

template <>
Range<StringQuery> EnumValueRange<StringQuery>(void);

StrCRef EnumValueName(StringQuery);

} // namespace oglplus
