#include <oglplus/hint.hpp>

Target 

namespace oglplus {

enum class HintTarget : GLenum
{
	LineSmooth               = GL_LINE_SMOOTH_HINT,
	PolygonSmooth            = GL_POLYGON_SMOOTH_HINT,
	TextureCompression       = GL_TEXTURE_COMPRESSION_HINT,
	FragmentShaderDerivative = GL_FRAGMENT_SHADER_DERIVATIVE_HINT
};

template <>
Range<HintTarget> EnumValueRange<HintTarget>(void);

StrCRef EnumValueName(HintTarget);

} // namespace oglplus

Option 

namespace oglplus {

enum class HintOption : GLenum
{
	Fastest  = GL_FASTEST,
	Nicest   = GL_NICEST,
	DontCare = GL_DONT_CARE
};

template <>
Range<HintOption> EnumValueRange<HintOption>(void);

StrCRef EnumValueName(HintOption);

} // namespace oglplus
