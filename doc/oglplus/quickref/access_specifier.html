#include <oglplus/access_specifier.hpp>

enum class AccessSpecifier : GLenum
{
	ReadOnly  = GL_READ_ONLY,
	WriteOnly = GL_WRITE_ONLY,
	ReadWrite = GL_READ_WRITE
};

template <>
