#include <oglplus/stencil_operation.hpp> 

enum class StencilOperation : GLenum
{
	Keep     = GL_KEEP,
	Zero     = GL_ZERO,
	Replace  = GL_REPLACE,
	Incr     = GL_INCR,
	Decr     = GL_DECR,
	Invert   = GL_INVERT,
	IncrWrap = GL_INCR_WRAP,
	DecrWrap = GL_DECR_WRAP
};

template <>
Range<StencilOperation> EnumValueRange<StencilOperation>(void);
