Precision type

#include <oglplus/precision_type.hpp> 

namespace oglplus {

enum class PrecisionType : GLenum
{
	LowFloat    = GL_LOW_FLOAT,
	MediumFloat = GL_MEDIUM_FLOAT,
	HighFloat   = GL_HIGH_FLOAT,
	LowInt      = GL_LOW_INT,
	MediumInt   = GL_MEDIUM_INT,
	HighInt     = GL_HIGH_INT
};

template <>
Range<PrecisionType> EnumValueRange<PrecisionType>(void);

StrCRef EnumValueName(PrecisionType);

} // namespace oglplus
