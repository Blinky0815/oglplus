Compare function

#include <oglplus/compare_function.hpp> 

enum class CompareFunction : GLenum
{
	LEqual   = GL_LEQUAL,
	GEqual   = GL_GEQUAL,
	Less     = GL_LESS,
	Greater  = GL_GREATER,
	Equal    = GL_EQUAL,
	NotEqual = GL_NOTEQUAL,
	Always   = GL_ALWAYS,
	Never    = GL_NEVER
};

template <>
Range<CompareFunction> EnumValueRange<CompareFunction>(void);

StrCRef EnumValueName(CompareFunction);
