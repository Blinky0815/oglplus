Color buffer

#include <oglplus/color_buffer.hpp> 

enum class ColorBuffer : GLenum
{
	None         = GL_NONE,
	FrontLeft    = GL_FRONT_LEFT,
	FrontRight   = GL_FRONT_RIGHT,
	BackLeft     = GL_BACK_LEFT,
	BackRight    = GL_BACK_RIGHT,
	Front        = GL_FRONT,
	Back         = GL_BACK,
	Left         = GL_LEFT,
	Right        = GL_RIGHT,
	FrontAndBack = GL_FRONT_AND_BACK
};

template <>
Range<ColorBuffer> EnumValueRange<ColorBuffer>(void);

StrCRef EnumValueName(ColorBuffer);
