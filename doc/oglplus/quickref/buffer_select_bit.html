#include <oglplus/buffer_select_bit.hpp> 

enum class BufferSelectBit : GLbitfield
{
	ColorBuffer   = GL_COLOR_BUFFER_BIT,
	DepthBuffer   = GL_DEPTH_BUFFER_BIT,
	StencilBuffer = GL_STENCIL_BUFFER_BIT
};

template <>
Range<BufferSelectBit> EnumValueRange<BufferSelectBit>(void);

StrCRef EnumValueName(BufferSelectBit);
